# Build from sources

# Build Cassandre

# Get the sources from Github

git clone git@github.com:cassandre-tech/cassandre-trading-bot.git

# Move to the source directory

cd cassandre-trading-bot

# Run the build

mvn package

# Build documentation

# Run local documentation

vuepress dev docs/src

Documentation web site is now running at http://0.0.0.0:8080/ (opens new window).

# Build static website

yarn --cwd docs build
